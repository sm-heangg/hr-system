//1. Step to Clone Project
git clone https://github.com/sm-heangg/hr-system.git

//2. to install vendor folder inside project to be able project can run artisan command
composer install

//3. rename file .env.example -> .env

//4. inside .env file change sqlite to mysql ( DB_CONNECTION=sqlite => DB_CONNECTION= mysql)
DB_CONNECTION= mysql
DB_HOST=127.0.0.1
DB_PORT=3306
DB_DATABASE= system_hr
DB_USERNAME=root
DB_PASSWORD=

//5. run command to generate key for project
php artisan key:generate

//6. run migration to create database
php artisan migrate

//7. create admin user for login
php artisan shield:super-admin
type 0 to select admin panel

//8. create admin user for login
php artisan shield:install
type 0 to select admin panel

//require command for project
//Install Filament
composer require filament/filament:"^4.0"

//QR Code Package
composer require simplesoftwareio/simple-qrcode "~4"

//permission for open camera on phone via browser first install it from microsoft store into your computer first before run command
ngrok http 8000


//for local cron job if employee not scan mark them as absent
php artisan schedule:work
